211  {
212  if (numArguments == N)
213  {
214  ossia::Vec<float, N> ret;

222  else if (vec_it->IsDouble())
223  ret.value[i] = vec_it->AsDoubleUnchecked();
224  else
225  return vec;
226 
227  i++;
228  }
229  return ret;
230  }
231  else
